## Import

```js
import Input from 'components/@ui/input'
```

## Usage

```jsx
<Input type='text' placeholder='Basic input'/>
```

## Variants

### Filled

```jsx
<Input 
  className='flex-1'
  type='text' 
  variant='filled' 
  placeholder='Outline'
/>
```

### Outline

```jsx
<Input 
  className='flex-1' 
  type='text' 
  variant='filled' 
  placeholder='Filled'
/>
```
