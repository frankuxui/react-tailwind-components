### Import

```js
import Avatar from 'components/@ui/avatar'
```

### Usage

```jsx
<Avatar image='/images/avatar.png' />
```

### Rounded

```jsx
<div className="flex items-center flex-wrap gap-2">
  <Avatar image='https://randomuser.me/api/portraits/men/35.jpg' rounded={true} />
</div>
```
